## Shared State ใน JavaScript vs Go

ใน JavaScript เนื่องจาก event loop เป็น single-threaded คุณแทบไม่ต้องกังวลเรื่อง race conditions บน shared data (ยกเว้นเมื่อใช้ SharedArrayBuffer + Worker Threads):

```typescript
let count = 0;
// ปลอดภัยใน Node.js เพราะ single-threaded
async function increment() { count++; }
await Promise.all([increment(), increment()]);
console.log(count); // 2 — ถูกต้องเสมอ
```

ใน Go goroutines รันบน threads จริง ดังนั้นการเข้าถึง shared variable พร้อมกันโดยไม่มี synchronization ทำให้เกิด **data race** ซึ่งเป็น undefined behavior:

```go
// อันตราย! Data race!
count := 0
var wg sync.WaitGroup
for i := 0; i < 100; i++ {
    wg.Add(1)
    go func() { defer wg.Done(); count++ }() // race!
}
wg.Wait()
// count อาจไม่เท่ากับ 100
```

## Mutex — Mutual Exclusion Lock

`sync.Mutex` ให้มีเพียง goroutine เดียวที่เข้าถึง critical section ได้ในเวลาหนึ่ง:

export const tsLockCode = `// TypeScript: ไม่จำเป็นต้องใช้ mutex ใน event loop ปกติ
// แต่ถ้าใช้ SharedArrayBuffer กับ Workers:
const sab = new SharedArrayBuffer(4);
const view = new Int32Array(sab);
Atomics.add(view, 0, 1); // atomic increment`;

export const goMutexCode = `// Go: sync.Mutex ปกป้อง shared state
package main

import (
    "fmt"
    "sync"
)

type SafeCounter struct {
    mu    sync.Mutex
    count int
}

func (c *SafeCounter) Increment() {
    c.mu.Lock()         // ล็อก
    defer c.mu.Unlock() // ปลดล็อกเมื่อ function return
    c.count++
}

func main() {
    var wg sync.WaitGroup
    counter := &SafeCounter{}

    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            counter.Increment()
        }()
    }
    wg.Wait()
    fmt.Println("Final count:", counter.count) // 100 เสมอ
}`;

## RWMutex — อ่านได้พร้อมกัน เขียนได้ครั้งเดียว

เมื่อ data ถูกอ่านบ่อยแต่เขียนน้อย ใช้ `sync.RWMutex`:

export const tsReadWriteCode = `// TypeScript: ไม่มี built-in RWMutex concept
// ใน Workers ต้องใช้ message passing แทน`;

export const goRWMutexCode = `// Go: RWMutex — readers พร้อมกัน, writer exclusive
package main

import (
    "fmt"
    "sync"
)

type Cache struct {
    mu   sync.RWMutex
    data map[string]string
}

func (c *Cache) Get(key string) string {
    c.mu.RLock()         // หลาย goroutines อ่านพร้อมกันได้
    defer c.mu.RUnlock()
    return c.data[key]
}

func (c *Cache) Set(key, value string) {
    c.mu.Lock()         // เขียน: exclusive lock
    defer c.mu.Unlock()
    c.data[key] = value
}`;

<GoOnly title="ตรวจหา Data Races ด้วย -race Flag">

Go มี built-in race detector ที่ทรงพลัง:

```bash
go run -race main.go
go test -race ./...
```

Race detector จะรายงาน data race ทันทีที่รันไทม์ detect ได้:

```
WARNING: DATA RACE
Write at 0x... by goroutine 7:
  main.main.func1()
Read at 0x... by goroutine 8:
  main.main.func1()
```

export const playgroundMutexCode = `package main

import (
	"fmt"
	"sync"
)

type SafeCounter struct {
	mu    sync.Mutex
	count int
}

func (c *SafeCounter) Increment() {
	c.mu.Lock()
	defer c.mu.Unlock()
	c.count++
}

func (c *SafeCounter) Value() int {
	c.mu.Lock()
	defer c.mu.Unlock()
	return c.count
}

func main() {
	var wg sync.WaitGroup
	counter := &SafeCounter{}

	for i := 0; i < 100; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			counter.Increment()
		}()
	}
	wg.Wait()
	fmt.Println("Final count:", counter.Value())
}`;
