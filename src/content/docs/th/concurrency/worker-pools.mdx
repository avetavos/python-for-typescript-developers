## Promise Pool vs Worker Pool

ใน TypeScript เมื่อต้องการจำกัดจำนวน concurrent operations คุณสร้าง Promise pool:

```typescript
async function processWithPool<T>(
  items: T[],
  concurrency: number,
  processor: (item: T) => Promise<void>
) {
  const pool: Promise<void>[] = [];
  for (const item of items) {
    const p = processor(item).then(() => {
      pool.splice(pool.indexOf(p), 1);
    });
    pool.push(p);
    if (pool.length >= concurrency) {
      await Promise.race(pool);
    }
  }
  await Promise.all(pool);
}
```

ใน Go worker pool ใช้ goroutines + channels ซึ่งเป็น pattern ที่ clean และมีประสิทธิภาพกว่า

## Worker Pool Pattern พื้นฐาน

export const tsPromisePoolCode = `// TypeScript: p-limit library สำหรับ concurrency control
import pLimit from 'p-limit';

const limit = pLimit(3); // max 3 concurrent
const tasks = [1, 2, 3, 4, 5, 6, 7, 8, 9];

const results = await Promise.all(
  tasks.map(n => limit(() => processTask(n)))
);`;

export const goWorkerPoolCode = `// Go: Worker Pool ด้วย goroutines + channels
package main

import (
    "fmt"
    "sync"
)

func worker(id int, jobs <-chan int, results chan<- int, wg *sync.WaitGroup) {
    defer wg.Done()
    for j := range jobs { // รับงานจาก jobs channel
        results <- j * j  // ส่งผลลัพธ์ไปยัง results channel
    }
}

func main() {
    const numWorkers = 3
    const numJobs = 9

    jobs := make(chan int, numJobs)
    results := make(chan int, numJobs)

    var wg sync.WaitGroup
    // launch workers
    for w := 1; w <= numWorkers; w++ {
        wg.Add(1)
        go worker(w, jobs, results, &wg)
    }

    // ส่งงาน
    for j := 1; j <= numJobs; j++ {
        jobs <- j
    }
    close(jobs) // แจ้งว่าไม่มีงานแล้ว

    wg.Wait()
    close(results)

    sum := 0
    for r := range results {
        sum += r
    }
    fmt.Println("Sum of squares:", sum) // 285
}`;

## Pipeline Pattern — ต่อ Workers เป็น Chain

export const tsPipelineCode = `// TypeScript: pipeline ด้วย async generators
async function* stage1(input: AsyncIterable<number>) {
  for await (const n of input) yield n * 2;
}
async function* stage2(input: AsyncIterable<number>) {
  for await (const n of input) yield n + 1;
}`;

export const goPipelineCode = `// Go: pipeline ด้วย channels
package main

import "fmt"

func generate(nums ...int) <-chan int {
    out := make(chan int, len(nums))
    for _, n := range nums {
        out <- n
    }
    close(out)
    return out
}

func double(in <-chan int) <-chan int {
    out := make(chan int, 10)
    go func() {
        for n := range in {
            out <- n * 2
        }
        close(out)
    }()
    return out
}

func main() {
    nums := generate(1, 2, 3, 4, 5)
    doubled := double(nums)
    for v := range doubled {
        fmt.Println(v) // 2, 4, 6, 8, 10
    }
}`;

<GoOnly title="เหตุใด Worker Pool ถึง Idiomatic ใน Go">

Go ออกแบบ concurrency primitives ให้เหมาะกับ worker pool pattern โดยธรรมชาติ:

- **Channel as queue** — jobs channel ทำหน้าที่เป็น concurrent-safe queue โดยอัตโนมัติ
- **for range on channel** — worker loop clean โดยไม่ต้องมี explicit loop control
- **close(jobs) เป็น signal** — ส่ง signal ว่างานหมดแล้วโดยไม่ต้องส่ง sentinel value
- **WaitGroup** — รอ workers ทั้งหมดเสร็จก่อน close results

Pattern นี้ scale ได้ดีมาก — เพิ่ม numWorkers เพื่อใช้ CPU cores มากขึ้น, ลดเพื่อจำกัด resource usage

</GoOnly>
