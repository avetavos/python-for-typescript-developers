## Handler signature

ใน Express handler รับ `(req, res, next)` ใน Echo ทุก handler มี signature ที่เป็นมาตรฐานเดียว:

```go
func(c echo.Context) error
```

`echo.Context` รวม request, response writer, path parameter, query parameter และ helper method ไว้ด้วยกัน คืนค่า `nil` เมื่อสำเร็จ หรือคืน `error` เพื่อให้ Echo จัดการ response ที่ fail

export const tsHandler = `// Express handler
app.get('/books/:id', (req, res, next) => {
  const id = req.params.id;
  const book = findBook(id);
  if (!book) return res.status(404).json({ message: 'not found' });
  res.json(book);
});`;

export const goHandler = `// Echo handler
func getBook(c echo.Context) error {
    id := c.Param("id")
    book, err := findBook(id)
    if err != nil {
        return echo.NewHTTPError(http.StatusNotFound, "not found")
    }
    return c.JSON(http.StatusOK, book)
}`;

## การลงทะเบียน route

Routing API ของ Echo คล้าย Express มาก เมธอดแมปตรงกับ HTTP verb

export const tsRoutes = `// Express
app.get('/books',      listBooks);
app.post('/books',     createBook);
app.get('/books/:id',  getBook);
app.put('/books/:id',  updateBook);
app.delete('/books/:id', deleteBook);`;

export const goRoutes = `// Echo
e.GET("/books",       listBooks)
e.POST("/books",      createBook)
e.GET("/books/:id",   getBook)
e.PUT("/books/:id",   updateBook)
e.DELETE("/books/:id", deleteBook)`;

<TsGo client:visible ts={tsRoutes} go={goRoutes} />

## Path parameter

`c.Param("name")` คืนค่า string ของ segment ที่มีชื่อ syntax ด้วย colon `:name` เหมือนกับ Express

export const tsParam = `// Express
app.get('/users/:id/posts/:postId', (req, res) => {
  const { id, postId } = req.params;
  res.json({ userId: id, postId });
});`;

export const goParam = `// Echo
e.GET("/users/:id/posts/:postId", func(c echo.Context) error {
    userID  := c.Param("id")
    postID  := c.Param("postId")
    return c.JSON(http.StatusOK, map[string]string{
        "userId": userID,
        "postId": postID,
    })
})`;

<TsGo client:visible ts={tsParam} go={goParam} />

## Query parameter

`c.QueryParam("name")` อ่านค่า query เดียว `c.QueryParams()` คืน `url.Values` (เป็น `map[string][]string`) สำหรับทั้งหมด

export const tsQuery = `// Express
app.get('/books', (req, res) => {
  const { page = '1', limit = '10' } = req.query;
  res.json({ page, limit });
});`;

export const goQuery = `// Echo
e.GET("/books", func(c echo.Context) error {
    page  := c.QueryParam("page")
    limit := c.QueryParam("limit")
    if page == "" { page = "1" }
    if limit == "" { limit = "10" }
    return c.JSON(http.StatusOK, map[string]string{
        "page": page, "limit": limit,
    })
})`;

<TsGo client:visible ts={tsQuery} go={goQuery} />

## ผูก handler เข้ากับ struct (แบบ NestJS controller)

NestJS controller ใช้ class ใน Go รูปแบบ idiomatic คือ struct ที่ method เป็น handler และ dependency injection เกิดขึ้นผ่าน field ของ struct

export const tsController = `// NestJS controller
@Controller('books')
export class BooksController {
  constructor(private readonly booksService: BooksService) {}

  @Get(':id')
  findOne(@Param('id') id: string) {
    return this.booksService.findOne(id);
  }
}`;

export const goController = `// Echo — handler struct (manual DI)
type BookHandler struct {
    store BookStore
}

func NewBookHandler(s BookStore) *BookHandler {
    return &BookHandler{store: s}
}

func (h *BookHandler) Get(c echo.Context) error {
    id := c.Param("id")
    book, err := h.store.FindByID(c.Request().Context(), id)
    if err != nil {
        return echo.NewHTTPError(http.StatusNotFound, err.Error())
    }
    return c.JSON(http.StatusOK, book)
}

// ลงทะเบียนใน main.go:
// bh := NewBookHandler(store)
// e.GET("/books/:id", bh.Get)`;

<GoOnly title="Wildcard และ route แบบลึกได้ไม่จำกัด">

Echo รองรับ wildcard segment ด้วย `*`:

```go
// จับ /static/css/app.css, /static/js/bundle.js ฯลฯ
e.GET("/static/*", staticHandler)

// ภายใน handler:
path := c.Param("*")   // "css/app.css"
```

ใช้สำหรับ serve static file หรือ proxy sub-path
