## Binding — อ่าน request body

ใน Express คุณเข้าถึง `req.body` (หลังจาก `express.json()`) ใน Echo, `c.Bind(&target)` ทำสิ่งเดียวกัน: อ่าน `Content-Type` header และ decode body เข้า struct ที่ส่งไป นอกจากนี้ยังสามารถ bind path param และ query param ได้หาก field ใน struct มี tag ที่ถูกต้อง

export const tsBinding = `// Express + TypeScript
interface CreateBookDTO {
  title: string;
  author: string;
  year: number;
}

app.post('/books', (req, res) => {
  const body = req.body as CreateBookDTO;
  // body.title, body.author, body.year พร้อมใช้
  res.status(201).json(body);
});`;

export const goBinding = `// Echo
type CreateBookRequest struct {
    Title  string \`json:"title"\`
    Author string \`json:"author"\`
    Year   int    \`json:"year"\`
}

func createBook(c echo.Context) error {
    var req CreateBookRequest
    if err := c.Bind(&req); err != nil {
        return echo.NewHTTPError(http.StatusBadRequest, err.Error())
    }
    // req.Title, req.Author, req.Year พร้อมใช้
    return c.JSON(http.StatusCreated, req)
}`;

## Struct tag — ระบบ annotation ของ Go

Struct tag คือ string literal ที่แนบกับ field ทำงานคล้าย TypeScript decorator แต่ไม่มีการ execute โค้ด — อ่านตอน runtime ผ่าน reflection `json:"name"` ควบคุม serialization; `validate:"rule"` อ่านโดย library validator

export const tsDecorators = `// TypeScript class-validator
import { IsString, IsInt, Min, IsNotEmpty } from 'class-validator';

class CreateBookDTO {
  @IsString()
  @IsNotEmpty()
  title: string;

  @IsString()
  author: string;

  @IsInt()
  @Min(1000)
  year: number;
}`;

export const goStructTags = `// Go struct tag
type CreateBookRequest struct {
    Title  string \`json:"title"  validate:"required"\`
    Author string \`json:"author" validate:"required"\`
    Year   int    \`json:"year"   validate:"required,min=1000"\`
}`;

## ลองใช้งาน — struct tag กับ encoding/json

Package `encoding/json` ของ stdlib อ่าน `json:"..."` tag ระหว่าง Marshal/Unmarshal รันได้ใน playground:

export const playgroundStructTags = `package main

import (
	"encoding/json"
	"fmt"
)

type Book struct {
	Title  string \`json:"title"\`
	Author string \`json:"author"\`
	Year   int    \`json:"year"\`
	// omitempty: field ถูกละเว้นจาก JSON เมื่อเป็น zero/empty
	ISBN   string \`json:"isbn,omitempty"\`
}

func main() {
	// Unmarshal (JSON -> struct)
	raw := \`{"title":"The Go Programming Language","author":"Donovan & Kernighan","year":2015}\`
	var b Book
	if err := json.Unmarshal([]byte(raw), &b); err != nil {
		fmt.Println("error:", err)
		return
	}
	fmt.Printf("Title:  %s\\n", b.Title)
	fmt.Printf("Author: %s\\n", b.Author)
	fmt.Printf("Year:   %d\\n", b.Year)
	fmt.Printf("ISBN:   %q (empty — ละเว้นใน output)\\n", b.ISBN)

	// Marshal (struct -> JSON)
	out, _ := json.MarshalIndent(b, "", "  ")
	fmt.Println(string(out))
}`;

## Validation ด้วย go-playground/validator

Echo ไม่มี validation built-in แต่เปิดเผย `e.Validator` วิธีมาตรฐานคือเชื่อมกับ `go-playground/validator`

export const tsClassValidator = `// NestJS global validation pipe
import { ValidationPipe } from '@nestjs/common';
app.useGlobalPipes(new ValidationPipe({ whitelist: true }));

// DTO
class CreateBookDTO {
  @IsString() @IsNotEmpty() title: string;
  @IsInt() @Min(1000) year: number;
}`;

export const goValidation = `// เชื่อม validator ครั้งเดียวใน main.go
import "github.com/go-playground/validator/v10"

type CustomValidator struct {
    v *validator.Validate
}

func (cv *CustomValidator) Validate(i any) error {
    return cv.v.Struct(i)
}

// ใน main():
e.Validator = &CustomValidator{v: validator.New()}

// Handler — bind แล้ว validate
func createBook(c echo.Context) error {
    var req CreateBookRequest
    if err := c.Bind(&req); err != nil {
        return echo.NewHTTPError(http.StatusBadRequest, err.Error())
    }
    if err := c.Validate(req); err != nil {
        return echo.NewHTTPError(http.StatusUnprocessableEntity, err.Error())
    }
    // req พร้อมใช้งานอย่างปลอดภัย
    return c.JSON(http.StatusCreated, req)
}`;

<GoOnly title="validate tag ที่ใช้บ่อย">

| Tag | ความหมาย |
|-----|---------|
| `required` | Field ต้องไม่เป็น zero value |
| `min=n` | ตัวเลข: ค่า ≥ n; string/slice: ความยาว ≥ n |
| `max=n` | ขอบเขตบน |
| `email` | รูปแบบ e-mail ที่ถูกต้อง |
| `url` | URL ที่ถูกต้อง |
| `oneof=a b c` | ค่าต้องเป็นหนึ่งในรายการที่ระบุ |
| `len=n` | ความยาวที่แน่นอน |

รวมกันด้วยเครื่องหมายจุลภาค: `validate:"required,min=1,max=200"`
