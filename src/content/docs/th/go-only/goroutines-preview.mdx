## JavaScript: concurrency model ด้วย async/await

JavaScript เป็น single-threaded concurrency มาจาก event loop: เมื่อ async operation (network, I/O) เสร็จ callback จะถูก schedule `async/await` คือ syntactic sugar บน Promise — ทำให้โค้ด async อ่านเหมือน synchronous แต่ยังคงมีแค่ thread เดียวที่รัน JavaScript ของผู้ใช้ในเวลาหนึ่ง

Go ใช้แนวทางที่แตกต่างอย่างสิ้นเชิง: **goroutine** คือ lightweight thread ที่ concurrent จริงๆ และ Go runtime schedule พวกมันบน OS thread หลายตัว คุณสามารถรัน goroutine ได้หลายล้านตัวบนเครื่องทั่วไป

## การเปิด goroutine

keyword `go` เปิด function ใน goroutine ใหม่ — execution context ที่ถูก schedule อย่างอิสระและเบาน้ำหนัก มัน return ทันที; goroutine รัน concurrently

export const tsAsyncCode = `// TypeScript — async/await (event loop, single thread)
async function fetchUser(id: number): Promise<User> {
  const res = await fetch(\`/users/\${id}\`);
  return res.json();
}

async function main() {
  // รัน sequential โดยค่าเริ่มต้น:
  const u1 = await fetchUser(1);
  const u2 = await fetchUser(2);

  // เพื่อรัน concurrent ใช้ Promise.all:
  const [u1, u2] = await Promise.all([fetchUser(1), fetchUser(2)]);
}`;

export const goGoroutineCode = `// Go — goroutine (OS-thread-backed, parallel จริงๆ)
func fetchUser(id int) User {
    // ... actual network call
    return User{ID: id}
}

func main() {
    // keyword go — เปิด fetchUser แบบ concurrent
    go fetchUser(1)   // fire and forget (ผลลัพธ์หาย)
    go fetchUser(2)

    // เพื่อรับผลลัพธ์ ใช้ channel หรือ sync.WaitGroup
}`;

## sync.WaitGroup — รอ goroutine เสร็จ

`sync.WaitGroup` คือวิธีที่ง่ายที่สุดในการรอ goroutine จำนวนที่รู้แน่นอนให้เสร็จก่อนที่ main goroutine จะออก คิดว่ามันเป็น counter ที่ block จนกว่าจะถึงศูนย์

export const tsPromiseAllCode = `// TypeScript — Promise.all รอทั้งหมด
const results = await Promise.all(
  [1, 2, 3].map(id => fetchUser(id))
);
console.log(results);`;

export const goWaitGroupCode = `func processItem(id int, wg *sync.WaitGroup) {
    defer wg.Done()                      // ลด counter เมื่อเสร็จ
    fmt.Printf("processing %d\\n", id)
}

func main() {
    var wg sync.WaitGroup

    for i := 1; i <= 3; i++ {
        wg.Add(1)                        // เพิ่ม counter
        go processItem(i, &wg)           // เปิด goroutine
    }

    wg.Wait()                            // block จนกว่า counter == 0
    fmt.Println("all done")
}`;

## Channel — สื่อสารระหว่าง goroutine

Channel คือ conduit ที่มี type สำหรับส่งค่าระหว่าง goroutine คุณ `send` ค่าด้วย `<-` และ `receive` ค่าด้วย `<-` Channel synchronize sender และ receiver — การ send block จนกว่าจะมีคนรับ และในทางกลับกัน (สำหรับ unbuffered channel)

export const tsChannelCode = `// TypeScript — ไม่มี channel primitive ใน native
// ใกล้ที่สุดคือ queue + event emitter หรือ stream
import { Readable } from "stream";
// ... ต้องการ plumbing ซับซ้อน`;

export const goChannelCode = `func producer(ch chan<- int) {
    for i := 0; i < 5; i++ {
        ch <- i          // ส่ง i เข้า channel
    }
    close(ch)            // signal ว่าไม่มีค่าอีกแล้ว
}

func main() {
    ch := make(chan int)  // unbuffered channel

    go producer(ch)      // รัน concurrent

    for v := range ch {  // รับจนกว่า channel ถูก close
        fmt.Println("received:", v)
    }
}`;

## ลองทำ

export const playgroundGoroutinesCode = `package main

import (
	"fmt"
	"sync"
)

func fetchData(id int, wg *sync.WaitGroup, results chan<- string) {
	defer wg.Done()
	// จำลองงาน
	result := fmt.Sprintf("result from worker %d", id)
	results <- result
}

func main() {
	const numWorkers = 3
	var wg sync.WaitGroup
	results := make(chan string, numWorkers) // buffered channel

	for i := 1; i <= numWorkers; i++ {
		wg.Add(1)
		go fetchData(i, &wg, results)
	}

	// ปิด channel เมื่อ goroutine ทั้งหมดเสร็จ
	go func() {
		wg.Wait()
		close(results)
	}()

	// รวบรวมผลลัพธ์
	for r := range results {
		fmt.Println(r)
	}
	fmt.Println("all workers done")
}`;

<GoOnly title="Goroutine ไม่ใช่ OS Thread">

OS thread ทั่วไปใช้ stack memory 1–8 MB Go runtime เริ่มต้นแต่ละ goroutine ด้วย stack เล็กๆ (~2–4 KB) ที่โตแบบ dynamic Go scheduler multiplexes goroutine หลายตัวบน OS thread pool ที่เล็กกว่า (M:N scheduling) หมายความว่า:

- **Goroutine หลายล้านตัว** สามารถอยู่ร่วมกันบนเครื่องทั่วไปได้
- การสร้าง goroutine ถูกมาก — microsecond และ KB ไม่กี่ KB
- การ block goroutine (เช่น รอ I/O) ไม่ block OS thread scheduler park goroutine และรันตัวอื่นแทน

เทียบกับ Node.js: JavaScript thread เดียวจัดการ callback ทั้งหมดผ่าน event loop Goroutine ของ Go รัน parallel จริงๆ บน multi-core hardware โดยไม่ต้องกำหนดค่าเพิ่มเติม

```go
// การเปิด 100,000 goroutine เป็นเรื่องปกติใน Go
for i := 0; i < 100_000; i++ {
    go func(id int) { /* lightweight work */ }(i)
}
```

</GoOnly>

<GoOnly title="keyword go และ Race Condition">

เพราะ goroutine รัน concurrent พวกมันจึงสามารถอ่านและเขียน memory เดิมพร้อมกัน นี่คือ **data race** — หนึ่งใน bug ที่อันตรายที่สุดใน concurrent programming — และ Go มีเครื่องมือตรวจจับ

```go
// DATA RACE — อย่าทำแบบนี้:
counter := 0
var wg sync.WaitGroup
for i := 0; i < 1000; i++ {
    wg.Add(1)
    go func() {
        defer wg.Done()
        counter++ // concurrent read-modify-write — RACE!
    }()
}
wg.Wait()

// รันด้วย: go run -race main.go
// race detector จะรายงานทันที
```

ทางเลือกที่ปลอดภัยคือ `sync.Mutex`, `sync/atomic`, หรือ channel Concurrency module เต็มรูปแบบครอบคลุม pattern ทั้งสามอย่างละเอียด

</GoOnly>
