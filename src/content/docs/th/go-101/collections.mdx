## Collections: จาก array ของ JS สู่ slice ของ Go

ใน JavaScript/TypeScript array เป็นแบบไดนามิกโดยปริยาย และ object ทำหน้าที่เป็น map ส่วน Go แยกเรื่องเหล่านี้ออกจากกัน: **arrays** ขนาดคงที่, **slices** แบบไดนามิก (ตัวที่คุณจะใช้ทุกวัน) และ **maps**

## Arrays เทียบกับ Slices

array ของ Go มีขนาดคงที่ฝังอยู่ใน type เลย คุณจะแทบไม่ได้ใช้มันตรงๆ **Slices** คือเวอร์ชันไดนามิกที่ยืดหยุ่น — มี array รองรับอยู่เบื้องหลังแต่มี length และ capacity ที่ขยายได้ตามต้องการ

export const tsArrayCode = `// TypeScript — always dynamic
const nums: number[] = [1, 2, 3];
nums.push(4);
console.log(nums.length);   // 4`;

export const goSliceCode = `// Go — slice (dynamic)
nums := []int{1, 2, 3}
nums = append(nums, 4)
fmt.Println(len(nums))  // 4`;

## Maps

maps ของ Go เหมือนกับ `Record<K,V>` หรือ `Map<K,V>` ของ TypeScript คีย์สามารถเป็น type ใดก็ได้ที่เปรียบเทียบกันได้ (comparable)

export const tsMapCode = `// TypeScript
const scores: Record<string, number> = {
    alice: 95,
    bob: 87,
};
scores["carol"] = 91;
delete scores["bob"];
console.log(scores["alice"]);`;

export const goMapCode = `// Go
scores := map[string]int{
    "alice": 95,
    "bob":   87,
}
scores["carol"] = 91
delete(scores, "bob")
fmt.Println(scores["alice"])`;

<TsGo client:visible ts={tsMapCode} go={goMapCode} />

export const playgroundCollCode = `package main

import (
	"fmt"
	"sort"
)

func main() {
	// Slice
	fruits := []string{"banana", "apple", "cherry"}
	fruits = append(fruits, "date")
	sort.Strings(fruits)
	fmt.Println("Fruits:", fruits)

	// Map
	count := map[string]int{}
	words := []string{"go", "is", "fun", "go", "is", "great", "go"}
	for _, w := range words {
		count[w]++
	}
	fmt.Println("Word counts:", count)
}`;

<GoOnly title="การตรวจสอบการมีอยู่ของคีย์ใน map">

ใน TypeScript การเข้าถึงคีย์ที่ไม่มีอยู่จะคืนค่า `undefined` แต่ใน Go การเข้าถึงคีย์ที่ไม่มีอยู่ใน map จะคืน **zero value** ของ type ของค่านั้น ในการแยกแยะระหว่าง "คีย์มีอยู่แต่ค่าเป็น zero value" กับ "คีย์ไม่มีอยู่" ให้ใช้รูปแบบสองค่า:

```go
scores := map[string]int{"alice": 0}

val := scores["bob"]         // val == 0, but is bob in the map?
val, ok := scores["alice"]   // val == 0, ok == true  (present!)
val, ok  = scores["bob"]     // val == 0, ok == false (absent)
```

สำนวน `ok` เป็น idiomatic ของ Go — คุณจะเห็นมันกับ type assertion และการรับค่าจาก channel ด้วย

</GoOnly>

<GoOnly title="make() สำหรับ slice และ map ที่จองหน่วยความจำล่วงหน้า">

เมื่อคุณรู้ขนาดล่วงหน้า ให้จองหน่วยความจำล่วงหน้าด้วย `make` เพื่อหลีกเลี่ยงการจัดสรรหน่วยความจำใหม่ซ้ำๆ:

```go
// slice of 10 ints, all zero
s := make([]int, 10)

// map pre-allocated for ~100 entries
m := make(map[string]int, 100)
```

</GoOnly>
